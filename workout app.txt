<script>
const workouts={
Monday:["Incline Bench","Dips","Chest Flies","Preacher Curls","Standing Cable Curls","Abs"],
Tuesday:["Lat Pulldowns","Upright Rows","Tricep Cable Pulldowns","Skull Crushers"],
Wednesday:["Seated Leg Extensions","Reverse Leg Curls","Calf Raises","Shrugs","Military Dumbbell Press","Cable Delt Flies","Cable Rear Delt Flies","Abs"],
Thursday:["Incline Bench","Chest Flies","Tricep Cable Pulldowns","Standing Cable Curls","Preacher Curls","Lat Pulldowns","Cuban Press","Cable Delt Flies"],
Friday:["Seated Leg Extensions","Reverse Leg Curls","Abductor Machine","Adductor Machine","Calf Raises","Abs"]
}

let currentDay="Monday"
let exerciseIndex=0
let restInterval

function saveData(key,value){
localStorage.setItem(key,JSON.stringify(value))
}

function loadData(key){
const d=localStorage.getItem(key)
return d?JSON.parse(d):{}
}

function renderButtons(){
const container=document.getElementById('dayButtons')
container.innerHTML=''
Object.keys(workouts).forEach(day=>{
const btn=document.createElement('button')
btn.innerText=day
btn.onclick=()=>{currentDay=day;exerciseIndex=0;renderExercises()}
container.appendChild(btn)
})
}

function renderExercises(){
const container=document.getElementById('exerciseList')
container.innerHTML=''

const saved=loadData('workoutData')

workouts[currentDay].forEach((ex,i)=>{
const card=document.createElement('div')
card.className='card'
if(i===exerciseIndex) card.style.border='3px solid #4CAF50'

const title=document.createElement('div')
title.innerText=ex

const inputs=document.createElement('div')
inputs.className='inputs'

const weight=document.createElement('input')
weight.placeholder='Weight'
weight.type='number'
weight.value=saved[ex]?.weight||''

const reps=document.createElement('input')
reps.placeholder='Reps'
reps.type='number'
reps.value=saved[ex]?.reps||''

weight.onchange=()=>{
saved[ex]={...saved[ex],weight:weight.value,date:new Date().toISOString()}
saveData('workoutData',saved)
updateChart()
}

reps.onchange=()=>{
saved[ex]={...saved[ex],reps:reps.value,date:new Date().toISOString()}
saveData('workoutData',saved)
}

inputs.appendChild(weight)
inputs.appendChild(reps)

card.appendChild(title)
card.appendChild(inputs)
container.appendChild(card)
})
}

function nextExercise(){
exerciseIndex++
if(exerciseIndex>=workouts[currentDay].length) exerciseIndex=0
renderExercises()
}

function startRest(seconds){
clearInterval(restInterval)
let remaining=seconds
const timer=document.getElementById('timer')
timer.innerText='Rest Timer: '+remaining

restInterval=setInterval(()=>{
remaining--
timer.innerText='Rest Timer: '+remaining
if(remaining<=0){
clearInterval(restInterval)
timer.innerText='Rest Finished ✔'
}
},1000)
}

let chart
function updateChart(){
const saved=loadData('workoutData')

const labels=[]
const data=[]

Object.keys(saved).forEach(ex=>{
if(saved[ex].weight){
labels.push(ex)
data.push(Number(saved[ex].weight))
}
})

const ctx=document.getElementById('progressChart')

if(chart) chart.destroy()

chart=new Chart(ctx,{type:'bar',data:{labels:labels,datasets:[{label:'Weight Lifted',data:data}]}})
}

renderButtons()
renderExercises()
updateChart()

</script>
